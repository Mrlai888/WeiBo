  5 upstream api {
  6    server  49.235.35.28:50001 max_fails=3 fail_timeout=30s;

  9 server {
 10     listen 80; // 监听端口
 11     server_name private.city.iot.com qcidebug-console.iothub.qq.com; // 域名 //可配置两个
 12 
 13    location /municipal_console {   //反向代理地  不会暴露服务ip
 14       proxy_pass http://api/municipal_console;  
 15    }
 16 
 17    location /cos/ {  // cos地址
 18      add_header Access-Control-Allow-Headers X-Requested-With,Authorization always;
 19      proxy_pass http://cos/;
 20    }
 21 
 22    location / {
 23         alias /data/cityfiles/;    //打包文件地址索引
 24         index index.html;       
 25         try_files $uri $uri/ /index.html;  //路径匹配
 26     }
